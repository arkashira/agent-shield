```python
import requests

class AgentOSClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://api.agentos.com"

    def register_tool(self, tool_name, tool_url):
        # Register the tool with the AgentOS API
        response = requests.post(
            f"{self.base_url}/tools",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={"name": tool_name, "url": tool_url},
        )
        response.raise_for_status()
        return response.json()["id"]

    def manage_memory(self, tool_name, memory_limit):
        # Manage memory for the tool
        response = requests.put(
            f"{self.base_url}/tools/{tool_name}",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={"memory_limit": memory_limit},
        )
        response.raise_for_status()
        return response.json()

    def set_security_policy(self, tool_name, policy):
        # Set security policy for the tool
        response = requests.put(
            f"{self.base_url}/tools/{tool_name}",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={"security_policy": policy},
        )
        response.raise_for_status()
        return response.json()
```

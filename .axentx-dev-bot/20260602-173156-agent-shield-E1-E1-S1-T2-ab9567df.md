## 1. Acceptance Criteria
1. **Auto‑load detection** – When `agent-shield.yaml` is present in the AgentOS config directory (`/etc/agentos/` by default), the Shield module is initialized on AgentOS startup without any code changes.
2. **Log confirmation** – On successful load, the system emits a log entry matching the regex:  
   `^\[INFO\] AgentOS Shield activated from .*agent-shield\.yaml$`.
3. **No startup failure** – The presence of a valid `agent‑shield.yaml` does **not** cause the AgentOS process to exit or throw uncaught exceptions.
4. **Graceful fallback** – If the file is absent or malformed, AgentOS starts normally and logs a warning: `AgentOS Shield configuration not found or invalid – proceeding without Shield`.
5. **YAML schema validation** – The file must conform to the documented Shield schema (e.g., top‑level keys `policy`, `rules`, optional `metadata`). Invalid schema triggers the warning in #4.
6. **Idempotent loading** – Re‑loading the same file (e.g., after a hot‑reload) does not duplicate log messages or create duplicate internal Shield instances.
7. **README accuracy** – The README’s “Zero‑Config” section contains the exact path and file name (`agent-shield.yaml`) and the expected log message format.

All criteria are measurable via automated tests or log inspection.

// __tests__/shieldLoader.unit.test.js
const fs = require('fs');
const path = require('path');
const { loadShieldConfig, validateShieldSchema } = require('../src/shield/loader');
const logger = require('../src/logger');

jest.mock('fs');
jest.mock('../src/logger');

describe('Shield Loader – Unit', () => {
  const CONFIG_DIR = '/etc/agentos';
  const SHIELD_FILE = path.join(CONFIG_DIR, 'agent-shield.yaml');

  afterEach(() => {
    jest.resetAllMocks();
  });

  test('loadShieldConfig reads file when it exists', () => {
    const fakeYaml = 'policy: strict\nrules:\n  - name: block‑malicious';
    fs.existsSync.mockReturnValue(true);
    fs.readFileSync.mockReturnValue(fakeYaml);

    const cfg = loadShieldConfig(CONFIG_DIR);

    expect(fs.readFileSync).toHaveBeenCalledWith(SHIELD_FILE, 'utf8');
    expect(cfg).toEqual({ policy: 'strict', rules: [{ name: 'block‑malicious' }] });
  });

  test('loadShieldConfig returns null and logs warning when file missing', () => {
    fs.existsSync.mockReturnValue(false);

    const cfg = loadShieldConfig(CONFIG_DIR);

    expect(cfg).toBeNull();
    expect(logger.warn).toHaveBeenCalledWith(
      expect.stringContaining('AgentOS Shield configuration not found')
    );
  });

  test('validateShieldSchema accepts a correct schema', () => {
    const good = { policy: 'lenient', rules: [] };
    expect(validateShieldSchema(good)).toBe(true);
  });

  test('validateShieldSchema rejects missing top‑level keys', () => {
    const bad = { rules: [] }; // missing policy
    expect(validateShieldSchema(bad)).toBe(false);
  });

  test('loadShieldConfig logs activation when schema valid', () => {
    const yaml = 'policy: strict\nrules: []';
    fs.existsSync.mockReturnValue(true);
    fs.readFileSync.mockReturnValue(yaml);
    // mock schema validator to return true
    jest.spyOn(require('../src/shield/loader'), 'validateShieldSchema').mockReturnValue(true);

    loadShieldConfig(CONFIG_DIR);

    expect(logger.info).toHaveBeenCalledWith(
      expect.stringMatching(/AgentOS Shield activated from .*agent-shield\.yaml$/)
    );
  });
});

```rust
use std::{
    sync::Arc,
    time::{SystemTime, UNIX_EPOCH},
};

use dashmap::DashMap;
use once_cell::sync::Lazy;
use serde::{Deserialize, Serialize};
use uuid::Uuid;

/// Truncated string helper (max 256 chars)
fn truncate(s: &str) -> String {
    if s.len() > 256 {
        format!("{}…", &s[..256])
    } else {
        s.to_string()
    }
}

/// In‑memory, thread‑safe store for traces
pub static TRACE_STORE: Lazy<DashMap<String, Trace>> = Lazy::new(DashMap::new());

/// JSON‑serialisable trace record
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Trace {
    pub trace_id: String,
    pub agent_id: String,
    pub tool_name: String,
    pub input: String,
    pub output: String,
    pub started_at: u128,
    pub finished_at: u128,
}

/// Context that lives for the duration of a *logical* tool call
/// (i.e. across all retries)
pub struct TraceContext {
    trace_id: String,
    agent_id: String,
    tool_name: String,
    started_at: u128,
}

impl TraceContext {
    /// Create a new trace (or reuse an existing one if supplied).
    /// The caller passes the agent id and tool name; the trace id is
    /// generated once and stored in the global map.
    pub fn new(agent_id: &str, tool_name: &str) -> Self {
        let trace_id = Uuid::new_v4().to_string();
        let started_at = SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap()
            .as_millis();

        let trace = Trace {
            trace_id: trace_id.clone(),
            agent_id: agent_id.to_string(),
            tool_name: tool_name.to_string(),
            input: String::new(),
            output: String::new(),
            started_at,
            finished_at: 0,
        };
        TRACE_STORE.insert(trace_id.clone(), trace);

        Self {
            trace_id,
            agent_id: agent_id.to_string(),
            tool_name: tool_name.to_string(),
            started_at,
        }
    }

    /// Finalise the trace after the last attempt (success or final failure).
    /// Truncates I/O and records the finish timestamp.
    pub fn finish(&self, input: &str, output: &str) {
        let finished_at = SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap()
            .as_millis();

        if let Some(mut trace) = TRACE_STORE.get_mut(&self.trace_id) {
            trace.input = truncate(input);
            trace.output = truncate(output);
            trace.finished_at = finished_at;
        }
    }

    /// Return the trace id for logging / propagation
    pub fn id(&self) -> &str {
        &self.trace_id
    }
}
```

```rust
#[cfg(test)]
mod tests {

        let mut trace = Trace::new("agent1", "toolX", "input");
        trace.start();
        trace.finish("output data");
        assert_eq!(trace.agent_id, "agent1");
        assert_eq!(trace.tool_name, "toolX");
        assert!(trace.input.len() <= 256);
        assert!(trace.output.len() <= 256);
        assert!(trace.start_ts < trace.end_ts);
    }

    #[test]
    fn trace_id_propagated_through_retry() {
        let mut ctx = TraceContext::new("agent1");
        let trace_id = ctx.start_trace("toolX", "input");
        // simulate retry
        let trace_id_retry = ctx.start_trace("toolX", "input");
        assert_eq!(trace_id, trace_id_retry);
    }

    #[test]
    fn json_logging_format() {
        let trace = Trace::new("agent1", "toolX", "input");
        let json = trace.to_json();
        let parsed: serde_json::Value = serde_json::from_str(&json).unwrap();
        assert_eq!(parsed["trace_id"], trace.id.to_string());
        assert_eq!(parsed["agent_id"], "agent1");
        assert_eq!(parsed["tool_name"], "toolX");
    }

    #[test]
```

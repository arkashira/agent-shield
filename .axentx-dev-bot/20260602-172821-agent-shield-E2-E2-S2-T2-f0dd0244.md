```rust
use axentx::agent::Agent;
use axentx::utils::log;
use axentx::utils::time;
use std::collections::HashMap;
use std::sync::{Arc, Mutex};

// Create a map to store agent memory usage and corresponding timestamps
lazy_static! {
    static ref AGENT_MEMORY_MAP: Arc<Mutex<HashMap<String, (u64, time::Instant)>>> = Arc::new(Mutex::new(HashMap::new()));
}

// Log agent cleanup events
pub fn log_agent_cleanup(agent_id: &str, timestamp: time::Instant) {
    let mut memory_map = AGENT_MEMORY_MAP.lock().unwrap();
    memory_map.insert(agent_id.to_string(), (0, timestamp));
    log::info!("Agent {} memory cleared at {}", agent_id, timestamp);
}

// Update memory usage metrics
pub fn update_memory_usage(agent_id: &str, memory_usage: u64) {
    let mut memory_map = AGENT_MEMORY_MAP.lock().unwrap();
    let (memory, timestamp) = memory_map.remove(agent_id).unwrap();
    log_agent_cleanup(agent_id, timestamp);
    log::info!("Agent {} memory usage updated to {}", agent_id, memory_usage);
}

// Clear agent memory after idle timeout
pub fn clear_agent_memory(agent_id: &str, timeout: u64) {
    let mut memory_map = AGENT_MEMORY_MAP.lock().unwrap();
    let timestamp = time::Instant::now();
    if let Some((memory, _timestamp)) = memory_map.get(agent_id) {
        if timestamp - _timestamp > timeout {
            log_agent_cleanup(agent_id, timestamp);
            log::info!("Agent {} memory cleared after {} seconds", agent_id, timeout);
            update_memory_usage(agent_id, 0);
            memory_map.remove(agent_id);
        }
    }
}
```

import time
import gc
from typing import List, Dict, Any
from collections import deque
import logging

logger = logging.getLogger(__name__)

class ContextPruner:
    def __init__(self, max_tokens: int = 1000, max_age_seconds: int = 300):
        self.max_tokens = max_tokens
        self.max_age_seconds = max_age_seconds
        self.conversation_history = deque()
        self.last_access_time = time.time()
        
    def add_message(self, message: Dict[str, Any]):
        """Add a message to conversation history"""
        self.conversation_history.append({
            'message': message,
            'timestamp': time.time()
        })
        self.last_access_time = time.time()
        
    def get_context(self) -> List[Dict[str, Any]]:
        """Get current context with pruning applied"""
        # Apply pruning based on token count or age
        self._prune_by_token_count()
        self._prune_by_age()
        
        return [item['message'] for item in self.conversation_history]
    
    def _prune_by_token_count(self):
        """Remove oldest messages until under token limit"""
        total_tokens = sum(len(str(msg.get('content', ''))) // 4 for msg in self.conversation_history)  # Approximate token count
        
        while total_tokens > self.max_tokens and len(self.conversation_history) > 0:
            removed = self.conversation_history.popleft()
            total_tokens -= len(str(removed['message'].get('content', ''))) // 4
            
    def _prune_by_age(self):
        """Remove messages older than max_age_seconds"""
        current_time = time.time()
        cutoff_time = current_time - self.max_age_seconds
        
        while (len(self.conversation_history) > 0 and 
               self.conversation_history[0]['timestamp'] < cutoff_time):
            self.conversation_history.popleft()
            
    def cleanup(self):
        """Perform garbage collection and cleanup"""
        gc.collect()
        logger.debug("Performed garbage collection")
        
    def get_memory_stats(self) -> Dict[str, Any]:
        """Get current memory statistics"""
        return {
            'conversation_length': len(self.conversation_history),
            'last_access_time': self.last_access_time,
            'total_tokens_estimate': sum(len(str(msg.get('content', ''))) // 4 for msg in self.conversation_history)
        }

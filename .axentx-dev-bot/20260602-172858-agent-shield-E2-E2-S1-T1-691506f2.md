# /opt/axentx/agent-shield/src/memory/context_pruner.py
import time
import tiktoken
from dataclasses import dataclass
from typing import List, Dict, Any

@dataclass
class Message:
    role: str
    content: str
    tokens: int

class ContextPruner:
    """Keeps conversation history within token & inactivity limits."""
    def __init__(self, max_tokens: int = 1_000, inactivity_timeout: int = 300):
        self.max_tokens = max_tokens
        self.inactivity_timeout = inactivity_timeout
        self.history: List[Message] = []
        self.last_activity = time.time()
        self._enc = tiktoken.get_encoding("cl100k_base")

    def add(self, role: str, content: str) -> None:
        tokens = len(self._enc.encode(content))
        self.history.append(Message(role, content, tokens))
        self.last_activity = time.time()
        self._prune_if_needed()

    def _prune_if_needed(self) -> None:
        total = sum(m.tokens for m in self.history)
        if total > self.max_tokens or (time.time() - self.last_activity) > self.inactivity_timeout:
            self._prune()

    def _prune(self) -> None:
        total = sum(m.tokens for m in self.history)
        while total > self.max_tokens and self.history:
            removed = self.history.pop(0)
            total -= removed.tokens
        self.last_activity = time.time()

    def stats(self) -> Dict[str, Any]:
        return {
            "total_tokens": sum(m.tokens for m in self.history),
            "history_len": len(self.history),
            "last_activity": self.last_activity,
        }

    def clear(self) -> None:
        self.history.clear()
        self.last_activity = time.time()

## 1. Acceptance Criteria
1. **Memory reduction** – After an agent finishes execution, the `/health/metrics` endpoint reports a **≥ 40 %** drop in resident memory (RSS) compared to the value recorded immediately before execution.  
2. **Conversation pruning** – Any conversation segment older than **1000 tokens** **or** idle for **≥ 5 minutes** is removed from the in‑memory context store.  
3. **Garbage‑collector schedule** – The internal GC task runs **every 5 minutes** and logs “GC run completed” when it finishes cleaning unreferenced objects.  
4. **Prometheus metrics** – The following metrics are exposed on the `/metrics` endpoint:  
   - `agent_shield_memory_bytes` (current process memory)  
   - `agent_shield_context_items` (number of retained conversation chunks)  
   - `agent_shield_gc_runs_total` (counter of GC executions)  
5. **No functional regression** – Existing agent behaviours (e.g., response generation, tool usage) remain unchanged after pruning logic is enabled.  
6. **Thread‑safety** – Concurrent agents may invoke the pruner without raising race‑condition exceptions or corrupting the context store.

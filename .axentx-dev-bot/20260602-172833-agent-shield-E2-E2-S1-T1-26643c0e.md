```python
# /opt/axentx/agent-shield/memory/monitor.py

import psutil
import gc
import os

# Set memory threshold (80% of total memory)
total_memory = psutil.virtual_memory().total
memory_threshold = total_memory * 0.8

def monitor_memory_usage():
    # Get current memory usage
    memory_usage = psutil.virtual_memory().used

    # Check if memory usage exceeds threshold
    if memory_usage > memory_threshold:
        # Trigger garbage collection
        gc.collect()

    # Return memory usage as a percentage
    return (memory_usage / total_memory) * 100

def get_real_time_memory_metrics():
    # Get real-time memory metrics
    memory_metrics = {
        'memory_usage': monitor_memory_usage(),
        'total_memory': total_memory,
        'memory_threshold': memory_threshold
    }

    return memory_metrics

# Example usage:
if __name__ == '__main__':
    memory_metrics = get_real_time_memory_metrics()
    print(f"Memory Usage: {memory_metrics['memory_usage']}%")
    print(f"Total Memory: {memory_metrics['total_memory']}")
    print(f"Memory Threshold: {memory_metrics['memory_threshold']}")
```

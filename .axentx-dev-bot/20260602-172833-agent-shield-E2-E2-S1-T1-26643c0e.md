def monitor_memory():
    while True:
        memory = psutil.virtual_memory()
        if memory.percent > 80:  # Trigger garbage collection when heap utilization > 80%
            tracemalloc.start()
            snapshot = tracemalloc.take_snapshot()
            top_stats = snapshot.statistics('lineno')

            print("[GC] Top 10 lines")
            for stat in top_stats[:10]:
                print(stat)

            print(f"[Warning] Memory usage is {memory.percent}%")

        time.sleep(60)  # Check memory usage every minute

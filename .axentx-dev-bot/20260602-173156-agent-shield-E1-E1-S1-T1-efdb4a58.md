## 1. Acceptance Criteria
1. **Automatic discovery** – When an `agent-shield.yaml` file is placed in the AgentOS configuration directory (`/opt/axentx/agent-os/config/`), the system automatically loads it during startup without any code changes.
2. **Schema validation** – The loader validates the YAML against the predefined schema (e.g., required keys: `shield_enabled`, `rules`, `logging_level`). Invalid schemas trigger a clear error message and halt startup.
3. **Runtime logging** – Upon successful load, the system writes a log entry at `INFO` level: `"AgentOS Shield activated with configuration: <summary>"`.
4. **Graceful fallback** – If the file is missing, the system continues to run with default Shield settings and logs a warning: `"agent-shield.yaml not found – using default Shield configuration"`.
5. **No side‑effects** – No other parts of AgentOS should be modified or recompiled; the loader must be a pure runtime component.
6. **Idempotence** – Re‑loading the same file (e.g., via hot‑reload) must not duplicate log entries or alter internal state beyond the intended update.

import pytest
from unittest.mock import patch, MagicMock
from agent_shield.config_loader import load_agent_shield_config, CONFIG_SCHEMA

@pytest.fixture
def valid_yaml(tmp_path):
    yaml_path = tmp_path / "agent-shield.yaml"
    yaml_path.write_text("""
shield_enabled: true
rules:
  - id: "rule1"
    action: "block"
logging_level: "debug"
""")
    return yaml_path

@pytest.fixture
def invalid_yaml(tmp_path):
    yaml_path = tmp_path / "agent-shield.yaml"
    yaml_path.write_text("""
shield_enabled: "yes"   # wrong type
rules: null
""")
    return yaml_path

def test_load_valid_config(valid_yaml):
    config = load_agent_shield_config(str(valid_yaml))
    assert config["shield_enabled"] is True
    assert len(config["rules"]) == 1
    assert config["logging_level"] == "debug"

def test_schema_validation_failure(invalid_yaml):
    with pytest.raises(ValueError, match="Schema validation failed"):
        load_agent_shield_config(str(invalid_yaml))

def test_missing_file(tmp_path):
    with patch("agent_shield.config_loader.os.path.exists", return_value=False):
        config = load_agent_shield_config("/nonexistent/path")
    assert config == CONFIG_SCHEMA.default()

def test_logging_on_success(monkeypatch):
    mock_logger = MagicMock()
    monkeypatch.setattr("agent_shield.config_loader.logger", mock_logger)
    load_agent_shield_config(str(valid_yaml))
    mock_logger.info.assert_called_with(
        pytest.match(r"AgentOS Shield activated with configuration: .*")
    )

def test_logging_on_missing(monkeypatch):
    mock_logger = MagicMock()
    monkeypatch.setattr("agent_shield.config_loader.logger", mock_logger)
    load_agent_shield_config("/nonexistent/path")
    mock_logger.warning.assert_called_with(
        "agent-shield.yaml not found – using default Shield configuration"
    )

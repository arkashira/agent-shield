**Implementation**

### gc_manager.py
```python
import psutil
import gc
import logging

class GCManager:
    def __init__(self):
        self.heap_utilization_threshold = 0.8
        self.memory_limit = 500 * 1024 * 1024  # 500MB

    def monitor_memory_usage(self):
        process = psutil.Process()
        memory_info = process.memory_info()
        heap_utilization = memory_info.rss / (1024 * 1024)  # Convert to MB
        return heap_utilization

    def trigger_gc(self):
        if self.monitor_memory_usage() > self.heap_utilization_threshold:
            logging.info("Triggering garbage collection")
            gc.collect()

    def get_memory_metrics(self):
        process = psutil.Process()
        memory_info = process.memory_info()
        return {
            "rss": memory_info.rss,
            "vms": memory_info.vms,
            "percent": memory_info.percent
        }
```

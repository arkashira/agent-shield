import pytest
from agentos_sdk.agentos_client import AgentOSClient

def test_register_and_list_tools():
    client = AgentOSClient()
    dummy_tool = lambda x: x
    client.register_tool("echo", dummy_tool)
    assert "echo" in client.list_tools()
    assert client._tool_registry["echo"] is dummy_tool

def test_memory_store_and_retrieve():
    client = AgentOSClient()
    client.memory_store("foo", 42)
    assert client.memory_retrieve("foo") == 42

def test_memory_retrieve_missing_key():
    client = AgentOSClient()
    with pytest.raises(KeyError):
        client.memory_retrieve("nonexistent")

def test_register_tool_overwrite():
    client = AgentOSClient()
    client.register_tool("dup", lambda x: x)
    client.register_tool("dup", lambda y: y + 1)
    assert client._tool_registry["dup"].__code__.co_freevars == ("y",)

def test_docstrings_present():

+import yaml

+class ToolCallValidator:

+    """
+
+    def __init__(self, config_path: str | Path | None = None):
+        """
+        Load the whitelist from the given config file.
+        If no path is provided, look for AGENT_SHIELD_CONFIG env var

+        """
+        import os
+

+            config_path = os.getenv("AGENT_SHIELD_CONFIG", "./config.yaml")
+
+        self.whitelist: List[str] = []
+        self._load_config(Path(config_path))
+
+    def _load_config(self, path: Path) -> None:
+        if not path.exists():
+            raise FileNotFoundError(f"Config file not found: {path}")
+        with path.open("r", encoding="utf-8") as f:
+            data = yaml.safe_load(f) or {}
+        self.whitelist = data.get("tool_whitelist", [])
+
+    def is_allowed(self, tool_name: str) -> bool:
+        """Return True if tool_name is in the whitelist."""
+        return tool_name in self.whitelist
+
+    def validate(self, tool_name: str) -> None:
+        """
+        Raise a ValueError if the tool is not allowed.
+        """
+        if not self.is_allowed(tool_name):
+            raise ValueError(f"Tool '{tool_name}' is not allowed by the whitelist.")

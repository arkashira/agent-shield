import logging
from typing import Dict, List

class RBACMiddleware:
    def __init__(self, roles: Dict[str, List[str]]):
        self.roles = roles
        self.logger = logging.getLogger(__name__)

    def check_access(self, user_id: str, tool_call_id: str, role: str):
        if user_id not in self.roles:
            self.logger.warning(f"User {user_id} does not have any roles")
            return False

        required_roles = self.roles[user_id]
        if role not in required_roles:
            self.logger.warning(f"User {user_id} does not have role {role} to invoke tool call {tool_call_id}")
            return False

        return True
```

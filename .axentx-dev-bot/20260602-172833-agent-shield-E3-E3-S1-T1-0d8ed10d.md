# memory_manager.py
from typing import Optional


class MemoryLimitExceeded(RuntimeError):
    """Raised when the agent tries to exceed its memory quota."""
    pass


class MemoryManager:
    """
    Simple token‑based memory manager.

    Parameters
    ----------
    limit : int
        Maximum number of tokens the agent may hold at any time.
    """

    def __init__(self, limit: int = 2048):
        if limit <= 0:
            raise ValueError("Memory limit must be > 0")
        self.limit = limit
        self._used = 0

    @property
    def used(self) -> int:
        """Current token usage."""
        return self._used

    @property
    def remaining(self) -> int:
        """Tokens still available."""
        return self.limit - self._used

    def allocate(self, amount: int) -> None:
        """Reserve ``amount`` tokens; raise if it would exceed the limit."""
        if amount < 0:
            raise ValueError("Allocation amount must be non‑negative")
        if self._used + amount > self.limit:
            raise MemoryLimitExceeded(
                f"Requested {amount} tokens, only {self.remaining} left"
            )
        self._used += amount

    def release(self, amount: int) -> None:
        """Free ``amount`` tokens; never goes below zero."""
        if amount < 0:
            raise ValueError("Release amount must be non‑negative")
        self._used = max(0, self._used - amount)

    def reset(self) -> None:
        """Return to a clean slate (useful between tool calls)."""
        self._used = 0

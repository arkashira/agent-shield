+import logging
+import time
+from functools import wraps
+from axentx.agent_shield.alert import alert
+
+def retry(max_attempts=3, backoff_factor=1):
+    def decorator(func):
+        @wraps(func)
+        def wrapper(*args, **kwargs):
+            attempt = 1
+            while attempt <= max_attempts:
+                try:
+                    return func(*args, **kwargs)
+                except Exception as e:
+                    logging.error(f"Attempt {attempt} failed: {str(e)}")
+                    if attempt < max_attempts:
+                        time.sleep(backoff_factor * (2 ** (attempt - 1)))
+                        attempt += 1
+                    else:
+                        alert("Ops channel", f"Failure after retries: {str(e)}")
+                        logging.error(f"Failure after retries: {str(e)}")
+                        return False
+        return wrapper
+    return decorator
+

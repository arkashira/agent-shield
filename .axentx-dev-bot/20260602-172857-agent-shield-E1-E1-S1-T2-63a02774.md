import logging
import time
import yaml

class CircuitBreaker:
    def __init__(self, config):
        self.config = config
        self.state = "closed"
        self.failure_count = 0
        self.last_failure_time = 0
        self.logger = logging.getLogger(__name__)

    def is_open(self):
        return self.state == "open"

    def try_execute(self, func):
        if self.is_open():
            self.logger.info("Circuit breaker is open, returning fallback response")
            return self.fallback_response()

        try:
            return func()
        except Exception as e:
            self.logger.error(f"Error executing function: {e}")
            self.failure_count += 1
            self.last_failure_time = time.time()
            if self.failure_count >= self.config["threshold"]:
                self.state = "open"
                self.logger.info("Circuit breaker tripped, blocking further calls for 30s")
                time.sleep(30)
            else:
                self.logger.info(f"Retrying in {self.get_backoff_time()}s")
                time.sleep(self.get_backoff_time())
                return self.try_execute(func)

    def get_backoff_time(self):
        return 2 ** self.failure_count

    def fallback_response(self):
        # Return cached response or mock data
        return "Fallback response"

    def reset(self):
        self.state = "closed"
        self.failure_count = 0
        self.last_failure_time = 0

> **Why this design?**  
> * The gauge is updated in a **daemon thread** so the main request‑handling thread is never blocked.  
> * The exporter runs on its own HTTP server (`start_http_server`) – this is the most common Prometheus pattern and keeps the FastAPI app lightweight.  
> * The `/metrics` route is still provided as a fallback for environments where the external exporter cannot be reached (e.g., container‑level port mapping).  
> * All code lives in a single module (`memory_metrics.py`) – no duplication, easy testing, and straightforward CI integration.

"""
Prometheus exporter for agent‑shield memory usage.

* Registers a Gauge `agent_shield_memory_bytes`.
* Starts a background thread that updates the gauge every 5 s.
* Exposes a FastAPI route (optional) that returns the current metrics.
"""

import threading
import time
import psutil
from prometheus_client import (
    Gauge,
    start_http_server,
    generate_latest,
    CONTENT_TYPE_LATEST,
)
from fastapi import Response

# ------------------------------------------------------------------
# 1️⃣  Gauge definition
# ------------------------------------------------------------------
AGENT_SHIELD_MEMORY_GAUGE = Gauge(
    "agent_shield_memory_bytes",
    "Resident Set Size (RSS) of the agent‑shield process (bytes)",
)

# ------------------------------------------------------------------
# 2️⃣  Background updater
# ------------------------------------------------------------------
def _update_memory_gauge() -> None:
    """Continuously update the gauge every 5 s."""
    process = psutil.Process()
    while True:
        try:
            AGENT_SHIELD_MEMORY_GAUGE.set(process.memory_info().rss)
        except Exception:  # pragma: no cover
            # Swallow to avoid crashing the thread
            pass
        time.sleep(5)


def start_memory_metrics_server(port: int = 9100) -> None:
    """
    Starts the Prometheus HTTP server and the background updater.
    """
    # 1. Start the HTTP server that exposes /metrics
    start_http_server(port)

    # 2. Start the updater thread
    thread = threading.Thread(
        target=_update_memory_gauge,
        daemon=True,
        name="mem-gauge-updater",
    )
    thread.start()


# ------------------------------------------------------------------
# 3️⃣  FastAPI helper (optional)
# ------------------------------------------------------------------
def get_prometheus_response() -> Response:
    """
    Return a FastAPI Response containing the latest metrics.
    This is useful if the exporter cannot be reached on its own port.
    """
    # Force an immediate gauge update so the response is fresh
    AGENT_SHIELD_MEMORY_GAUGE.set(psutil.Process().memory_info().rss)
    return Response(
        content=generate_latest(),
        media_type=CONTENT_TYPE_LATEST,
    )

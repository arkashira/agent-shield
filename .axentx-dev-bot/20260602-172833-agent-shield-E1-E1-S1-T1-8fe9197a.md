const fs = require('fs').promises;
const path = require('path');

class MemoryStorage {
  constructor(storagePath = './data/memory') {
    this.storagePath = storagePath;
    this.ensureDirectory();
  }

  async ensureDirectory() {
    try {
      await fs.access(this.storagePath);
    } catch {
      await fs.mkdir(this.storagePath, { recursive: true });
    }
  }

  async save(item) {
    const filePath = path.join(this.storagePath, `${item.id}.json`);
    await fs.writeFile(filePath, JSON.stringify(item, null, 2));
  }

  async get(id) {
    const filePath = path.join(this.storagePath, `${id}.json`);
    try {
      const data = await fs.readFile(filePath, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      if (error.code === 'ENOENT') return null;
      throw error;
    }
  }

  async find(query) {
    const items = [];
    try {
      const files = await fs.readdir(this.storagePath);
      for (const file of files) {
        if (!file.endsWith('.json')) continue;
        
        const filePath = path.join(this.storagePath, file);
        const data = await fs.readFile(filePath, 'utf8');
        const item = JSON.parse(data);
        
        // Apply filters
        if (query.agentId && item.agentId !== query.agentId) continue;
        if (query.type && item.type !== query.type) continue;
        if (query.fromTimestamp && item.timestamp < query.fromTimestamp) continue;
        if (query.toTimestamp && item.timestamp > query.toTimestamp) continue;
        
        items.push(item);
      }
      
      // Sort by timestamp descending
      items.sort((a, b) => b.timestamp - a.timestamp);
      
      // Apply pagination
      const start = query.offset || 0;
      const end = start + (query.limit || items.length);
      return items.slice(start, end);

import json
from typing import Any, Dict


class ValidationError(RuntimeError):
    """Raised when a tool‑call payload does not meet the required schema."""
    pass


class ToolCallValidator:
    """
    Simple validator that ensures a tool‑call dictionary contains the required
    fields and that the ``args`` value is JSON‑serialisable.

    Expected schema::

        {
            "tool": "<tool‑name>",   # required, non‑empty string
            "args": { ... }          # required, JSON‑serialisable mapping
        }
    """

    REQUIRED_KEYS = {"tool", "args"}

    @staticmethod
    def _is_json_serializable(value: Any) -> bool:
        """Return ``True`` if *value* can be round‑tripped through ``json.dumps``."""
        try:
            json.dumps(value)
            return True
        except (TypeError, OverflowError):
            return False

    def validate(self, payload: Dict[str, Any]) -> None:
        """
        Validate *payload* according to the schema above.

        :param payload: The tool‑call dictionary to validate.
        :raises ValidationError: If any rule is violated.
        """
        if not isinstance(payload, dict):
            raise ValidationError("Payload must be a dictionary.")

        missing = self.REQUIRED_KEYS - payload.keys()
        if missing:
            raise ValidationError(f"Missing required keys: {sorted(missing)}")

        # ``tool`` must be a non‑empty string
        tool = payload["tool"]
        if not isinstance(tool, str) or not tool.strip():
            raise ValidationError("`tool` must be a non‑empty string.")

        # ``args`` must be JSON‑serialisable
        args = payload["args"]
        if not self._is_json_serializable(args):
            raise ValidationError("`args` must be JSON‑serialisable.")

from typing import Callable, Any, Dict, Optional

class SecurityMiddleware:
    """
    Intercepts and validates agent tool calls to enforce security policies.
    Prevents injection attacks and policy violations before execution.
    """

    def __init__(self, policies: Optional[Dict[str, Callable]] = None):
        """
        Initialize the middleware with a dictionary of tool names to policy functions.
        
        Args:
            policies: A mapping of tool_name -> validation_function.
        """
        self.policies = policies or {}

    def intercept(self, tool_name: str, arguments: Dict[str, Any]) -> bool:
        """
        Validates the tool call against registered policies.
        
        Args:
            tool_name: The name of the tool being called.
            arguments: The arguments passed to the tool.
            
        Returns:
            bool: True if the call is safe, False if blocked.
        """
        if tool_name not in self.policies:
            # Default allow if no policy is defined
            return True
            
        try:
            return self.policies[tool_name](arguments)
        except Exception:
            # Fail-safe: block on any policy error
            return False

    def register_policy(self, tool_name: str, policy_func: Callable) -> None:
        """
        Register a new security policy for a specific tool.
        
        Args:
            tool_name: The name of the tool.
            policy_func: A function that takes arguments and returns bool.
        """
        self.policies[tool_name] = policy_func
```

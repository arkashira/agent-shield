import os
import time
from collections import deque
from typing import Deque, Tuple

from fastapi import APIRouter, Depends, Header, HTTPException, status

router = APIRouter()

# --- Metrics Storage ---
_CALLS: Deque[Tuple[float, bool, float]] = deque()  # (timestamp, success, latency_ms)
WINDOW_SECONDS = 5 * 60  # 5-minute window

def _purge_old_entries() -> None:
    """Remove entries older than the sliding window."""
    cutoff = time.time() - WINDOW_SECONDS
    while _CALLS and _CALLS[0][0] < cutoff:
        _CALLS.popleft()

def record_tool_call(success: bool, latency_ms: float) -> None:
    """Record a tool call's outcome and latency."""
    _CALLS.append((time.time(), success, latency_ms))
    _purge_old_entries()

def _aggregate() -> Tuple[int, int, float]:
    """Returns (total_calls, failed_calls, avg_latency_ms) for the current window."""
    _purge_old_entries()
    total = len(_CALLS)
    if total == 0:
        return 0, 0, 0.0
    failures = sum(1 for _, ok, _ in _CALLS if not ok)
    avg_latency = sum(lat for _, _, lat in _CALLS) / total
    return total, failures, avg_latency

# --- API-Key Auth ---
API_KEY_HEADER = "X-API-Key"
EXPECTED_API_KEY = os.getenv("HEALTH_API_KEY", "changeme")  # Override in production

def _require_api_key(x_api_key: str = Header(..., alias=API_KEY_HEADER)) -> bool:
    if x_api_key != EXPECTED_API_KEY:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid API key",
        )
    return True

# --- Endpoint ---
UPTIME_THRESHOLD = 99.8  # percent

@router.get(
    "/health/tool-calls",
    dependencies=[Depends(_require_api_key)],
    response_model=dict,
    tags=["health"],
)
def health_tool_calls() -> dict:
    """Return health metrics for tool calls."""
    total, failures, avg_latency = _aggregate()
    success = total - failures
    uptime = (success / total * 100) if total else 100.0

    payload = {
        "total_calls": total,
        "failed_calls": failures,
        "avg_latency_ms": round(avg_latency, 2),
        "uptime_percent": round(uptime, 2),
    }

    if uptime < UPTIME_THRESHOLD:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail=f"Uptime below required {UPTIME_THRESHOLD}%",
            headers={"X-Uptime": f"{uptime:.2f}"},
        )

    return payload
```

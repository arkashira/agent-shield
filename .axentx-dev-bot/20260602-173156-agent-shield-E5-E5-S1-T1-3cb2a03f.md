import os
from functools import wraps
import psutil
from flask import Flask, render_template, request, jsonify, abort

app = Flask(__name__, static_folder="static", template_folder="templates")

# ------------------------------------------------------------------
# 1️⃣  Auth helper
# ------------------------------------------------------------------
API_KEY = os.getenv("DASHBOARD_API_KEY", "demo-key")  # set in env

def require_api_key(f):
    """Require X‑API‑KEY header to match env var."""
    @wraps(f)
    def decorated(*args, **kwargs):
        key = request.headers.get("X-API-KEY")
        if key != API_KEY:
            abort(401, description="Invalid API key")
        return f(*args, **kwargs)
    return decorated

# ------------------------------------------------------------------
# 2️⃣  Metrics source – replace with real store later
# ------------------------------------------------------------------
def get_metrics():
    """Return a dict of metrics. Swap out for real data source."""
    # Dummy data – replace with real queries / telemetry
    return {
        "total_calls": 12345,
        "success_rate": 0.97,
        "avg_latency_ms": 250,
        "memory_per_agent_mb": psutil.virtual_memory().percent,  # placeholder
    }

# ------------------------------------------------------------------
# 3️⃣  Routes
# ------------------------------------------------------------------
@app.route("/dashboard")
@require_api_key
def dashboard():
    """Render the dashboard page."""
    return render_template("dashboard.html")

@app.route("/api/metrics")
@require_api_key
def api_metrics():
    """Return JSON metrics."""
    return jsonify(get_metrics())

# ------------------------------------------------------------------
# 4️⃣  Run

    # For dev: allow host/port override
    host = os.getenv("FLASK_HOST", "0.0.0.0")
    port = int(os.getenv("FLASK_PORT", 5000))
    app.run(host=host, port=port, debug=True)

1. **Endpoint & Routing** – `GET /dashboard` returns **HTTP 200** and renders the HTML dashboard template.  
2. **Authentication** – Requests must include a valid `X‑API‑Key` header; missing or invalid keys return **HTTP 401** with JSON `{ "error": "Invalid API key" }`.  
3. **Metric Display** – Rendered page contains the four metric fields **Total Calls**, **Success Rate**, **Average Latency**, **Memory Usage**, each showing a numeric value.  
4. **Refresh Interval** – The page includes client‑side JavaScript that automatically re‑fetches metric data **every 30 seconds** (`30000 ms`).  
5. **Data Source** – Dashboard obtains metric JSON from internal endpoint `GET /api/metrics` (same API‑key protection) and populates the placeholders correctly.  
6. **Error‑Handling UI** – If `/api/metrics` returns a non‑200 status or malformed JSON, the dashboard displays a banner: *“Unable to load metrics – please try again later.”*  
7. **Responsive Layout** – Using a responsive grid (e.g., Bootstrap), the dashboard renders without horizontal overflow on viewports **≥ 320 px**.

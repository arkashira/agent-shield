import time
import random

def call_with_retry(func, max_retries=3, initial_backoff=0.5):
    """
    Calls the given function with retry mechanism.

    Args:
    - func: The function to be called.
    - max_retries: The maximum number of retries.
    - initial_backoff: The initial back-off time in seconds.

    Returns:
    - The result of the function call if successful, otherwise None.
    """
    backoff = initial_backoff
    for attempt in range(max_retries + 1):
        try:
            # Attempt to call the function
            result = func()
            # If successful, return the result
            return result
        except Exception as e:
            # If not successful, check if we've reached the max retries
            if attempt < max_retries:
                # Calculate the back-off time with jitter
                backoff_with_jitter = backoff * (1 + random.uniform(0, 0.1))
                # Sleep for the back-off time
                time.sleep(backoff_with_jitter)
                # Exponentially increase the back-off time
                backoff *= 2
            else:
                # If all retries fail, raise the last exception
                raise e
    # If we've reached this point, all retries have failed
    return None

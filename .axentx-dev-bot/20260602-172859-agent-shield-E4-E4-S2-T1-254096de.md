"""Prometheus metrics for agent-shield.

This module ONLY defines collectors and helpers. It does NOT start any
HTTP server — the metrics are exposed via the Flask app's /metrics route
defined in src/app.py.
"""
from __future__ import annotations

from prometheus_client import Counter, Histogram, Gauge, REGISTRY, generate_latest

# --- Collectors (default REGISTRY) -----------------------------------------

TOOL_CALLS = Counter(
    "agent_shield_tool_calls_total",
    "Total tool calls routed through agent-shield.",
    ["tool", "status"],
)

TOOL_CALL_DURATION = Histogram(
    "agent_shield_tool_call_duration_seconds",
    "End-to-end tool call latency in seconds.",
    ["tool"],
    buckets=(0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0),
)

ACTIVE_AGENTS = Gauge(
    "agent_shield_active_agents",
    "Number of currently active agents.",
)

MEMORY_HITS = Counter(
    "agent_shield_memory_hits_total",
    "Memory cache hits/misses.",
    ["result"],
)


def metrics_response() -> tuple[bytes, str]:
    """Return (payload_bytes, content_type) for the /metrics endpoint."""
    return generate_latest(REGISTRY), "text/plain; version=0.0.4; charset=utf-8"


def record_tool_call(tool: str, status: str, duration_seconds: float) -> None:
    """Convenience helper used by call sites."""
    TOOL_CALLS.labels(tool=tool, status=status).inc()
    TOOL_CALL_DURATION.labels(tool=tool).observe(duration_seconds)
```

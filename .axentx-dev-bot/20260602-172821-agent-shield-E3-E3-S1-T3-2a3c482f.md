```rust
use serde_json::json;
use std::fs::{OpenOptions, File};
use std::io::Write;
use chrono::Utc;

pub fn redact_sensitive_fields(log_entry: &mut serde_json::Value) {
    if let Some(api_key) = log_entry.get("api_key") {
        log_entry["api_key"] = json!("REDACTED");
    }
}

pub fn log_tool_call(agent_id: &str, tool_name: &str, input: &str, output: &str, status: &str) {
    let mut log_entry = json!({
        "agent_id": agent_id,
        "tool_name": tool_name,
        "input": truncate(input, 1024),
        "output": truncate(output, 1024),
        "timestamp": Utc::now().to_rfc3339(),
        "status": status,
    });

    redact_sensitive_fields(&mut log_entry);

    let log_file_path = "/var/log/agent-shield/tool_calls.log";
    let mut log_file = OpenOptions::new()
        .append(true)
        .create(true)
        .open(log_file_path)
        .expect("Unable to open log file");

    writeln!(log_file, "{}", log_entry.to_string()).expect("Unable to write to log file");
}

fn truncate(data: &str, max_length: usize) -> String {
    if data.len() > max_length {
        data[..max_length].to_string()
    } else {
        data.to_string()
    }
}
```

use std::time::{SystemTime, UNIX_EPOCH};

// Define a struct to hold log information
#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct Log {
    agent_id: String,
    tool_name: String,
    input: String,
    output: String,
    timestamp: String,
    status: String,
}

/// Logs a message with full context, ensuring sensitive fields are redacted.
pub fn log_message(agent_id: &str, tool_name: &str, input: &str, output: &str, status: &str) -> io::Result<()> {
    // Create a new log entry
    let log = Log {
        agent_id: agent_id.to_string(),
        tool_name: tool_name.to_string(),
        input: input.to_string(),
        output: output.to_string(),
        timestamp: get_timestamp(),
        status: status.to_string(),
    };

    // Redact sensitive fields BEFORE writing to disk
    let redacted_log = redact_sensitive_fields(&log);
    let log_json = serde_json::to_string(&redacted_log)
        .map_err(|e| io::Error::new(io::ErrorKind::Other, e))?;

    // Ensure the log directory exists
    let log_dir = PathBuf::from("/var/log/agent-shield");
    fs::create_dir_all(&log_dir)?;

    // Append the redacted log entry to a daily log file
    let log_file = log_dir.join("agent.log");
    let mut file = OpenOptions::new()
        .create(true)
        .append(true)
        .open(log_file)?;
        
    writeln!(file, "{}", log_json)?;

    Ok(())
}

/// Helper function to redact sensitive fields from a Log struct
fn redact_sensitive_fields(log: &Log) -> Log {
    let mut redacted_log = log.clone();
    redacted_log.input = redact_string(&log.input);
    redacted_log.output = redact_string(&log.output);
    redacted_log
}

        .replace("password", "[REDACTED]")
}

}

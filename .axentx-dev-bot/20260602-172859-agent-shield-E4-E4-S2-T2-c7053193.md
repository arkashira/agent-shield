# Example protected endpoint (optional – demonstrates usage)
# ----------------------------------------------------------------------
class ToolCall(BaseModel):
    tool: str
    args: dict

# Initialise a default shield with a whitelist; in production this could be
# loaded from config or environment variables.
default_shield = AgentShield(allowed_tools=["search", "summarize"])

@app.post("/call")
def call_tool(payload: ToolCall):
    """
    Validate the incoming tool call using AgentShield.
    Returns 200 if the call is allowed, otherwise 400.
    """
    try:
        default_shield.check_call(payload.dict())
        return {"result": "call accepted"}
    except ShieldViolationError as exc:
        raise HTTPException(status_code=400, detail=str(exc))
    except ValidationError as exc:
        raise HTTPException(status_code=422, detail=exc.errors())

pub fn detect_memory_leaks(memory: Arc<Memory>) {
    let mut high_usage = false;
    let mut start_time = std::time::Instant::now();

    loop {
        thread::sleep(Duration::from_secs(1));

        let usage = memory.get_usage();
        if usage > memory.limit * 0.9 {
            high_usage = true;
            start_time = std::time::Instant::now();
        } else if high_usage && start_time.elapsed().as_secs() >= 300 {
            warn!("Memory leak detected with stack trace");
            // Log stack trace
            high_usage = false;
        }
    }
}

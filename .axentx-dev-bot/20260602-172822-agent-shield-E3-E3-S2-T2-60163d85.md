use std::env;
use base64;
use aes_gcm::{Aes256Gcm, Key, Nonce};

struct Config {
    memory_encryption_key: String,
}

impl Config {
    fn new() -> Self {
        let memory_encryption_key = env::var("MEMORY_ENCRYPTION_KEY")
            .expect("MEMORY_ENCRYPTION_KEY environment variable must be set")
            .into_bytes();

        // Validate key length
        if memory_encryption_key.len() != 32 {
            panic!("MEMORY_ENCRYPTION_KEY must be 32 bytes long");
        }

        // Validate key format
        let base64_key = base64::encode(&memory_encryption_key);
        if base64_key.len() != 44 {
            panic!("MEMORY_ENCRYPTION_KEY must be a 32-byte base64-encoded string");
        }

        Self {
            memory_encryption_key: String::from_utf8(memory_encryption_key).unwrap(),
        }
    }

    fn encrypt(&self, data: &[u8]) -> Result<Vec<u8>, aes_gcm::Error> {
        let key = Key::from_slice(&self.memory_encryption_key.as_bytes());
        let nonce = Nonce::from_slice(&[0; 12]);
        let cipher = Aes256Gcm::new(key);
        let encrypted = cipher.encrypt(nonce, data)?;
        Ok(encrypted)
    }

    fn decrypt(&self, encrypted: &[u8]) -> Result<Vec<u8>, aes_gcm::Error> {
        let key = Key::from_slice(&self.memory_encryption_key.as_bytes());
        let nonce = Nonce::from_slice(&[0; 12]);
        let cipher = Aes256Gcm::new(key);
        let decrypted = cipher.decrypt(nonce, encrypted)?;
        Ok(decrypted)
    }
}

"""
Retry decorator with exponential back‑off, logging, and Ops alerting.

All behaviour is fully configurable via keyword arguments.
"""

import time
import logging
import functools
from typing import Callable, Any, Tuple, Type

# ------------------------------------------------------------------
# Logger
# ------------------------------------------------------------------
_logger = logging.getLogger("agent_shield.retry")

# ------------------------------------------------------------------
# Ops alert helper
# ------------------------------------------------------------------
def alert_ops(message: str) -> None:
    """
    Stub for sending an Ops alert.

    In production this should post to Slack, PagerDuty, etc.
    For unit tests the function is monkey‑patched.
    """
    _logger.error(f"[OPS ALERT] {message}")

# ------------------------------------------------------------------
# Retry decorator
# ------------------------------------------------------------------
def retry(
    *,
    max_attempts: int = 3,
    base_delay: float = 1.0,
    backoff_factor: float = 2.0,
    allowed_exceptions: Tuple[Type[BaseException], ...] = (Exception,),
    alert_on_failure: bool = True,
) -> Callable[[Callable[..., Any]], Callable[..., Any]]:
    """
    Decorator that retries a function call on specified exceptions.

    Parameters
    ----------
    max_attempts : int
        Total number of attempts (initial call + retries).  Default 3.
    base_delay : float
        Initial delay in seconds before the first retry.  Default 1.0.
    backoff_factor : float
        Multiplier for exponential back‑off.  Default 2.0.
    allowed_exceptions : tuple[type, ...]
        Exception types that trigger a retry.  Default (Exception,).
    alert_on_failure : bool
        Whether to call :func:`alert_ops` after the final failure.  Default True.
    """
    def decorator(func: Callable[..., Any]) -> Callable[..., Any]:
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            attempt = 0
            while attempt < max_attempts:
                try:
                    return func(*args, **kwargs)
                except allowed_exceptions as exc:
                    attempt += 1
                    _logger.warning(
                        "Attempt %s/%s failed for %s: %s",
                        attempt,
                        max_attempts,
                        func.__name__,
                        exc,
                    )
                    if attempt == max_attempts:
                        # Final failure – log and alert
                        _logger.error(
                            "All %s attempts failed for %s: %s",
                            max_attempts,
                            func.__name__,
                            exc,
                        )
                        if alert_on_failure:
                            alert_ops(
                                f"Tool call {func.__name__} failed after "
                                f"{max_attempts} attempts: {exc}"
                            )
                        raise
                    # Exponential back‑off
                    delay = base_delay * (backoff_factor ** (attempt - 1))
                    time.sleep(delay)
        return wrapper

import pytest
from unittest import mock
from agent_shield.retry import retry_decorator, OpsAlert

# Helper: fake tool that fails N times then succeeds
def make_tool(fail_times, return_value="ok"):
    calls = {"count": 0}
    def tool(*args, **kwargs):
        if calls["count"] < fail_times:
            calls["count"] += 1
            raise RuntimeError(f"fail {calls['count']}")
        return return_value
    return tool

@pytest.fixture
def mock_logger(monkeypatch):
    logger = mock.Mock()
    monkeypatch.setattr("agent_shield.retry.logger", logger)
    return logger

@pytest.fixture
def mock_alert(monkeypatch):
    alert = mock.Mock(spec=OpsAlert)
    monkeypatch.setattr("agent_shield.retry.ops_alert", alert)
    return alert

def test_success_on_first_try(mock_logger):
    tool = make_tool(fail_times=0)
    decorated = retry_decorator(tool)
    result = decorated()
    assert result == "ok"
    mock_logger.info.assert_not_called()
    mock_logger.error.assert_not_called()

def test_retry_success_after_failures(mock_logger, mock_alert):
    tool = make_tool(fail_times=2)  # fails twice, succeeds on 3rd
    decorated = retry_decorator(tool)
    result = decorated()
    assert result == "ok"
    # 2 error logs + 1 success log
    assert mock_logger.error.call_count == 2
    assert mock_logger.info.call_count == 1
    mock_alert.send.assert_not_called()

def test_retry_exhaustion_and_alert(mock_logger, mock_alert):
    tool = make_tool(fail_times=5)  # always fails
    decorated = retry_decorator(tool)
    with pytest.raises(RuntimeError):
        decorated()
    assert mock_logger.error.call_count == 3  # 3 retries
    mock_alert.send.assert_called_once()
    # Verify alert payload contains last exception details
    alert_payload = mock_alert.send.call_args[0][0]
    assert "fail 3" in alert_payload

def test_exponential_backoff(monkeypatch, mock_logger):
    sleeps = []
    monkeypatch.setattr("time.sleep", lambda t: sleeps.append(t))
    tool = make_tool(fail_times=3)
    decorated = retry_decorator(tool)
    with pytest.raises(RuntimeError):
        decorated()
    assert sleeps == [1, 2, 4]  # back‑off sequence

def test_timeout_per_attempt(monkeypatch, mock_logger):
    # Simulate a hanging call that times out after 0.1s
    def hanging(*args, **kwargs):
        time.sleep(1)
    tool = hanging
    decorated = retry_decorator(tool, timeout=0.1, max_retries=1)
    with pytest.raises(TimeoutError):
        decorated()
    assert mock_logger.error.call_count == 1

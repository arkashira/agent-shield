use aes_gcm::{Aes256Gcm, Key, Nonce};
use aes_gcm::aead::{Aead, NewAead};
use base64::{engine::general_purpose, DecodeError};
use rand::Rng;
use std::env;
use thiserror::Error;

#[derive(Error, Debug)]
pub enum EncryptionError {
    #[error("Invalid base64 in encryption key")]
    Base64(#[from] DecodeError),
    #[error("Invalid key length, must be 32 bytes")]
    InvalidKeyLength,
    #[error("AES-GCM decryption failed")]
    DecryptionFailed,
    #[error("Missing encryption key in environment")]
    KeyMissing,
}

pub struct MemoryEncryption {
    key: Key<Aes256Gcm>,
}

impl MemoryEncryption {
    pub fn new() -> Result<Self, EncryptionError> {
        let key_env = env::var("MEMORY_ENCRYPTION_KEY")
            .map_err(|_| EncryptionError::KeyMissing)?;

        let key_bytes = general_purpose::STANDARD.decode(key_env)?;
        if key_bytes.len() != 32 {
            return Err(EncryptionError::InvalidKeyLength);
        }

        Ok(Self {
            key: Key::<Aes256Gcm>::from_slice(&key_bytes),
        })
    }

    pub fn encrypt(&self, plaintext: &[u8]) -> Result<(Vec<u8>, Nonce<Aes256Gcm>), EncryptionError> {
        let cipher = Aes256Gcm::new(&self.key);
        let nonce = Aes256Gcm::generate_nonce(&mut rand::thread_rng());
        let ciphertext = cipher.encrypt(&nonce, plaintext.as_ref())
            .map_err(|_| EncryptionError::DecryptionFailed)?;
        Ok((ciphertext, nonce))
    }

    pub fn decrypt(&self, ciphertext: &[u8], nonce: &Nonce<Aes256Gcm>) -> Result<Vec<u8>, EncryptionError> {
        let cipher = Aes256Gcm::new(&self.key);
        cipher.decrypt(nonce, ciphertext)
            .map_err(|_| EncryptionError::DecryptionFailed)
    }
}

## 1. Acceptance Criteria
- **Encryption**: Agent memory files (`*.mem`) are stored encrypted with AES‑256‑GCM.  
- **Key Source**: The key is read from the environment variable `MEMORY_ENCRYPTION_KEY`, which must be a 32‑byte value encoded in base64.  
- **Decryption**: Decryption occurs only in memory during agent runtime; no plaintext is ever written to disk.  
- **Performance**: Encryption/decryption adds < 5 % overhead compared to the baseline memory‑write/read times measured on a typical agent workload.  
- **Error Handling**:  
  - Missing or malformed key → initialization error.  
  - Corrupted ciphertext → decryption error, agent aborts with clear message.  
- **Compatibility**: Existing memory format is preserved except for the added ciphertext wrapper; older agents should fail gracefully if they encounter encrypted files without the proper key.

## 3. Integration Tests

| Test | Description | Expected Outcome |
|------|-------------|------------------|
| **Happy Path – Write & Read** | Agent writes a memory snapshot to disk, then reads it back during the same run. | File exists, size > 0, decrypted content matches original. |
| **Happy Path – Persistence** | Agent writes memory, restarts, reads memory again. | Decrypted content matches original; no plaintext file left on disk. |
| **Happy Path – Performance** | Measure average write/read time over 100 iterations; compare to baseline (no encryption). | Overhead < 5 %. |
| **Edge – Missing Key** | Start agent with `MEMORY_ENCRYPTION_KEY` unset. | Agent fails to start with clear error. |
| **Edge – Corrupted Ciphertext** | Corrupt a byte in the stored file, then start agent. | Decryption error, agent aborts with clear message. |
| **Edge – Wrong Key on Restart** | Write memory with key A, restart agent with key B. | Decryption fails, agent aborts. |
| **Edge – Large Payload** | Write a 10 MB memory snapshot. | Encryption/decryption completes, size on disk ~ payload + overhead, performance within limits. |

*Implementation notes:*  
- Use a test harness that spawns the agent process with controlled env vars.  
- Capture stdout/stderr for error messages.  
- Use `std::time::Instant` for performance measurement.

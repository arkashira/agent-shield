```python
#!/usr/bin/env python3
import os
import psutil
import time
import logging
import sys

logging.basicConfig(level=logging.INFO)

def get_memory_usage():
    process = psutil.Process(os.getpid())
    return process.memory_info().rss

def main():
    limit = int(os.getenv('AGENT_MEMORY_LIMIT', '512m').replace('m', '')) * 1024 * 1024  # convert to bytes
    threshold = 0.8 * limit
    high_usage_count = 0
    high_usage_duration = 0
    start_time = time.time()

    while True:
        usage = get_memory_usage()
        if usage > threshold:
            high_usage_count += 1
            high_usage_duration = time.time() - start_time
            logging.warning(f"Memory usage {usage/1024/1024:.2f}MB exceeds 80% of limit {limit/1024/1024}MB. Sustained high usage for {high_usage_duration:.2f}s.")
            if high_usage_duration > 300:  # 5 minutes
                logging.error("Memory leak detected after 5 minutes of sustained high usage. Generating stack trace.")
                import traceback
                traceback.print_stack()
                break
        else:
            high_usage_count = 0
            start_time = time.time()
        time.sleep(1)

if __name__ == "__main__":
    main()
```

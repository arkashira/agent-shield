import psutil
import logging
import time
from config import MEMORY_LIMIT, MEMORY_THRESHOLD, ALERT_CHANNEL

logger = logging.getLogger(__name__)

def get_memory_usage():
    process = psutil.Process()
    return process.memory_info().rss / (1024 * 1024)  # in MB

def get_top_memory_functions():
    import line_profiler
    profiler = line_profiler.LineProfiler()
    profiler.add_function(get_memory_usage)
    profiler.run("get_memory_usage()")
    profiler.print_stats()
    return profiler.get_line_stats()[:5]

def run_profiler():
    memory_usage = get_memory_usage()
    if memory_usage > MEMORY_THRESHOLD * psutil.virtual_memory().total / (1024 * 1024):
        logger.warning(f"Memory usage exceeded {MEMORY_THRESHOLD * 100}% threshold")
        # Send alert to Ops channel
        send_alert(ALERT_CHANNEL, f"Memory usage exceeded {MEMORY_THRESHOLD * 100}% threshold")
    top_functions = get_top_memory_functions()
    logger.info(f"Top 5 memory-heavy functions: {top_functions}")
    # Log top 5 memory-heavy functions
    log_top_functions(top_functions)

def schedule_profiler():
    while True:
        run_profiler()
        time.sleep(600)  # 10 minutes

def main():
    schedule_profiler()

if __name__ == "__main__":
    main()
```

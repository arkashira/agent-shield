const AGENT_MEMORY_LIMIT: u64 = 512 * 1024 * 1024; // 512MB by default

}

let agent_memory_usage_gauge = register_int_gauge!(

).unwrap();
let memory_usage = get_memory_usage();
agent_memory_usage_gauge.set(memory_usage as u64);

// Implement memory leak detection and logging
let mut previous_memory_usage = get_memory_usage();
let leak_threshold = memory_limit / 10;
let mut leak_duration = std::time::Instant::now();

loop {
    let current_memory_usage = get_memory_usage();
    if current_memory_usage > previous_memory_usage + leak_threshold {
        let elapsed_time = std::time::Instant::now().duration_since(leak_duration);
        if elapsed_time.as_secs() > 300 { // 5 minutes
            log::error!("Memory leak detected: {}", current_memory_usage);
            backtrace::print();
            break;
        }
    }
    previous_memory_usage = current_memory_usage;

}

## 1. Acceptance Criteria

| # | Criterion | Measurable Target |
|---|-----------|-------------------|
| 1 | **Memory cap** | Each agent process cannot exceed 512 MB of resident set size (RSS) unless `AGENT_MEMORY_LIMIT` is set to a higher value. |
| 2 | **Warning log** | When an agent’s RSS > 80 % of its limit, a log entry with level `WARN` and key `memory_warning` is emitted. |
| 3 | **Prometheus metric** | A gauge metric `agent_memory_usage_bytes` is exposed per agent (label `agent_id`) and reflects the current RSS. |
| 4 | **Leak detection** | If an agent’s RSS stays > 90 % of its limit for ≥ 5 minutes, a log entry with level `ERROR` and key `memory_leak` is emitted, including a stack trace. |
| 5 | **Configurable limit** | The environment variable `AGENT_MEMORY_LIMIT` (in MB) overrides the default 512 MB. |
| 6 | **Graceful shutdown** | When an agent exceeds its limit, it receives a `SIGTERM` and exits after flushing logs. |

// Helper to simulate memory usage

fn test_env_override() {

    assert_eq!(cfg.limit_bytes, 1024 * 1024 * 1024);
}

#[test]
fn test_warning_emitted_at_80_percent() {
    let mut tracker = MemoryTracker::new("agent1", 512 * 1024 * 1024);
    tracker.update(fake_rss(410 * 1024 * 1024)); // 80.2%
    assert!(tracker.warned);
}

#[test]
fn test_no_warning_below_80_percent() {
    let mut tracker = MemoryTracker::new("agent2", 512 * 1024 * 1024);
    tracker.update(fake_rss(300 * 1024 * 1024));
    assert!(!tracker.warned);
}

#[test]
fn test_leak_detection_after_5_minutes() {
    let mut tracker = MemoryTracker::new("agent3", 512 * 1024 * 1024);
    for _ in 0..5 {
        tracker.update(fake_rss(470 * 1024 * 1024)); // 91.8%
        tracker.advance(Duration::from_secs(60));
    }
    assert!(tracker.leak_detected);
}

#[test]
fn test_leak_not_detected_if_fluctuating() {
    let mut tracker = MemoryTracker::new("agent4", 512 * 1024 * 1024);
    for _ in 0..3 {

        tracker.advance(Duration::from_secs(60));
    }
    tracker.update(fake_rss(300 * 1024 * 1024)); // drop below 90%
    for _ in 0..3 {
        tracker.update(fake_rss(470 * 1024 * 1024));

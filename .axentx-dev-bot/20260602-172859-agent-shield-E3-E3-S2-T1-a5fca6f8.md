# agent_shield/security.py
import datetime
import hashlib
import json
from typing import Tuple


class SafetyShield:
    """
    Combines:
    * Input validation
    * Timestamp + integrity hash
    * Keyword‑based safety filtering
    """

    DEFAULT_DISALLOWED = ["kill", "destroy", "erase", "delete", "terminate"]

    def __init__(self, disallowed_keywords: list | None = None):
        self.disallowed_keywords = disallowed_keywords or self.DEFAULT_DISALLOWED

    # ------------------------------------------------------------------
    # 1️⃣  Validation & safety
    # ------------------------------------------------------------------
    def _validate_input(self, data: dict) -> None:
        if not isinstance(data, dict):
            raise ValueError("Input data must be a dictionary")

    def _is_safe(self, content: str) -> Tuple[bool, str]:
        """Return (allowed, reason)."""
        lowered = content.lower()
        for kw in self.disallowed_keywords:
            if kw in lowered:
                return False, f"Disallowed keyword detected: '{kw}'"
        return True, "Message is safe"

    # ------------------------------------------------------------------
    # 2️⃣  Integrity helpers
    # ------------------------------------------------------------------
    def _generate_integrity_hash(self, data: dict) -> str:
        data_str = json.dumps(data, sort_keys=True)
        return hashlib.sha256(data_str.encode()).hexdigest()

    # ------------------------------------------------------------------
    # 3️⃣  Public API
    # ------------------------------------------------------------------
    def protect(self, data: dict) -> dict:
        """
        Validate, hash, timestamp, and safety‑check the payload.

        Returns a dict containing:
        - original content
        - timestamp (ISO‑8601)
        - integrity_hash
        - allowed (bool)
        - reason (str)
        """
        self._validate_input(data)

        content = data.get("content", "")
        allowed, reason = self._is_safe(content)

        protected = {
            "content": content,
            "timestamp": datetime.datetime.utcnow().isoformat() + "Z",
            "integrity_hash": self._generate_integrity_hash(data),
            "allowed": allowed,
            "reason": reason,
        }
        return protected
